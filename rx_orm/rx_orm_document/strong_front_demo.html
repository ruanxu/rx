插入一条person数据 对比后端 

    <script src="/Scripts/rx.js"></script>
    <script src="/Scripts/rx_manager.js"></script>
    <script>
        //方式1
        new person({
            //id: 1,
            name: "jack",
            age: 18,
            gender: "男"
        }).insert_entity(function(data){ console.info(data); });
         
        //方式2
        //insert_or_update_entity这种情况id不为0或者不为null就进行更新操作，否则还是添加
        new person({
            //id: 1,
            name: "jack",
            age: 18,
            gender: "男"
        }).insert_or_update_entity(function(data){ console.info(data); });
         
        //方式3
        //这种情况在web开发中经常使用，非常方便
        //request_fill方法直接可以将请求中的参数填充到实体中
        //前端中request_fill只能接受url地址栏的显示参数
        new person().request_fill().insert_or_update_entity(function (data) {
            console.info(data);
        });
    </script>

修改一个person的信息 对比后端 

    <script src="/Scripts/rx.js"></script>
    <script src="/Scripts/rx_manager.js"></script>
    <script>
        //方式1
        //将id为1的person对象的name修改为tom
        //如果is_use_null为true，会将id为1的person对象的除name以外的字段全部修改为null
        new person({
            id: 1,
            name: "tom"
            //is_use_null: true
        }).update_entity(function (data) { console.info(data); });
     
        //方式2
        //简化了方式1的操作
        //request_fill方法在前端只能接受url地址栏的显示参数
        new person().request_fill().update_entity(function (data) { console.info(data); });
     
        //方式3
        //将gender不为"女"且age在"16，17，18"中的person的name修改为"tom"
        //set_[属性名]_field方法会在where_keys中加入这个字段，但是不会重复加入
        //set_[属性名]_field方法就是弱类型set_rx_field的包装版，使用更加精简
        new person({ gender: "女", name: "tom", age: "16,17,18" })
            .set_gender_field({ compare: compare_symbol.not_equal })
            .set_age_field({ compare: compare_symbol.contain })
            .update_entity(function (data) { console.info(data); });
    </script>
