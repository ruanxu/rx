    <script>
        //方式1
        var obj = new rx_entity("person",
        {
            //id : 1,
            name: "jack",
            age: 18,
            gender: '男'
        });
        rx_manager.insert_entity(function (data) {
            console.info(data);
        }, obj);
     
        //方式2
        //insert_or_update_entity这种情况id不为0或者不为null就进行更新操作，否则还是添加
        var obj = new rx_entity("person",
        {
            id: 1,
            name: "jack",
            age: 18,
            gender: '男'
        });
        rx_manager.insert_or_update_entity(function (data) {
            console.info(data);
        }, obj);
     
        //方式3
        //这种情况在web开发中经常使用，非常方便
        //request_fill方法直接可以将请求中的参数填充到实体中
        //前端中request_fill方法只能接受url地址的显示传值
        //弱类型rx_entity在没有任何rx_filed成员时使用request_fill方法是无效的
        rx_manager.insert_or_update_entity(function (data) {
            console.info(data);
        }, new rx_entity("person", { name: "", age: "", gender: "" }).request_fill());
     
        //方式4
        //这个一看就懂，就不用做解释了，不限于insert
        //前端orm中对于dml操作需要接口开放对应权限才能使用
        rx_manager.execute_non_query(function (data) {
            console.info(data);
        }, "insert into person values('jack',18,'男')");
    </script>

添加一个部门，然后添加一个人属于这个部门 对比后端 

    <script src="/Scripts/rx.js"></script>
    <script src="/Scripts/rx_manager.js"></script>
    <script>
        //方式1
        var dept = new rx_entity("dept");
        dept.add("name", "研发部");
        rx_manager.insert_entity(function (data) {
            //添加成功后，data的tag中会out自增的id
            var pre = new rx_entity("person");
            pre.Add("name", "张三");
            pre.Add("age", 18);
            pre.Add(gender, "男");
            pre.Add("dept_id", data.tag);
            rx_manager.insert_entity(function (data) {
                console.info(data);
            }, pre);
        }, dept)
        //这样看似可行，但是有个缺点，分别执行两次insert，没有使用事务机制整体执行，所以不可行
     
        //方式2
        //对象sql语句拼接方式
        //弱类型rx_entity在没有任何rx_filed成员时使用request_fill方法是无效的
        var full_sql_string = " declare @id int = 0 ";
     
        var dept = new rx_entity("dept", {
            name: "研发部"
        });
     
        //将新的部门对象转换成insert语句
        full_sql_string += " " + rx_manager.transaction_insert_string_build(dept) + " ";
     
        full_sql_string += " set @id = @@identity ";
     
        var pre = new rx_entity("person", {
            name: "张三",
            age: 18,
            gender: "男",
            dept_id: "@id"
        })
        pre.dept_id.build_quote = false;//dept_id在进行sql生成操作时不会生成单引号(')
        //将新的人员对象转换成insert语句
        full_sql_string += " " + rx_manager.transaction_insert_string_build(pre) + " ";
     
        rx_manager.transaction_execute_non_query(function (data) {
            console.info(data)
        }, full_sql_string);
        //最终执行的sql如下所示
        //declare @id int = 0
        //insert into dept values('研发部')
        //set @id = @@identity
        //insert into person values('张三', 18, '男', @id)
     
        //方式3
        //进一步简化了方式2的操作
        //弱类型rx_entity在没有任何rx_filed成员时使用request_fill方法是无效的
     
        var full_sql_string = " declare @id int = 0 ";
     
        //将新的部门对象转换成insert语句
        full_sql_string += " " + rx_manager.transaction_insert_string_build(new rx_entity("dept", { name: "" }).request_fill()) + " ";
     
        full_sql_string += " set @id = @@identity ";
     
        //将新的人员对象转换成insert语句
        full_sql_string += " " + rx_manager.transaction_insert_string_build(new rx_entity("person",
                                         {
                                              name: "",
                                              age: "",
                                              gender: "",
                                              dept_id: ""
                                         })
                                         .request_fill()
                                         .set_rx_field("dept_id",
                                         {
                                              build_quote: false,
                                              value: "@id"
                                         })) + " ";
     
        rx_manager.transaction_execute_non_query(function (data) {
            console.info(data)
        }, full_sql_string);
        //整体操作起来还是比较繁琐，但是使用强类型开发能够进一步简化
    </script>
修改一个person的信息 对比后端 

    <script src="/Scripts/rx.js"></script>
    <script src="/Scripts/rx_manager.js"></script>
    <script>
        //方式1
        //将id为1的person对象的name修改为tom
        var obj = new rx_entity("person", {
            id: 1,
            name: "tom"
        });
        //如果is_use_null为true，会将id为1的person对象的除name以外的字段全部修改为null
        //obj.is_use_null = true;//is_use_null默认为false
        rx_manager.update_entity_by_id(function (data) {
            console.info(data);
        }, obj);
     
        //方式2
        //简化了方式1的操作
        //弱类型rx_entity在没有任何rx_filed成员时使用request_fill方法是无效的
        rx_manager.update_entity_by_id(function (data) {
            console.info(data);
        }, new rx_entity("person", { name: "", age: "", gender: "" }).request_fill());
     
        //方式3
        //将person表中年龄为16，17，18的人员的name修改为tom，且性别不等于"女"
        var obj = new rx_entity("person");
        obj.Add("age", "16,17,18");
        //指定age字段的比较运算符为 in
        obj.age.compare_symbol = compare_symbol.contain;
        obj.Add("gender", "女");
        //指定gender字段的比较运算符为不等于 !=
        obj.gender.compare_symbol = compare_symbol.not_equal;
        obj.Add("name", "tom");
     
        //指定参与条件运算的字段名称，可以多次指定
        obj.set_where_keys("age", "gender");
        rx_manager.update_entity_by_where_keys(function (data) {
            console.info(data);
        }, obj);
     
        //方式4
        //简化了方式3
        //弱类型rx_entity在没有任何rx_filed成员时使用request_fill方法是无效的
        rx_manager.update_entity_by_where_keys(function (data) {
            console.info(data);
        },
            new rx_entity("person", { age: "", gender: "", name: "" }).request_fill()
            .set_rx_field("age", { compare: compare_symbol.contain })
            .set_rx_field("gender", { compare: compare_symbol.not_equal })
        );
        //set_rx_field方法会在where_keys中加入这个字段，但是不会重复加入
    </script>
