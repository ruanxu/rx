修改一个person的信息 对比前端 

    using models;//这个namespace是你自己定义的实体类的namespace
    using rx;
     
    //方式1
    //将id为1的person对象的name修改为tom
    //如果is_use_null为true，会将id为1的person对象的除name以外的字段全部修改为null
    dml_result result = new person()
    {
        id = 1,
        name = "tom"
        //is_use_null = true
    }.update_entity();
     
    //方式2
    //简化了方式1的操作
    dml_result result = new person().request_fill().update_entity();
            
    //方式3
    //将gender不为"女"且age在"16，17，18"中的person的name修改为"tom"
    //set_[属性名]_field方法会在where_keys中加入这个字段，但是不会重复加入
    //set_[属性名]_field方法就是弱类型set_rx_field的包装版，使用更加精简
    dml_result result = new person(){ gender = "女", name = "tom"}
        .set_gender_field(compare: compare_symbol.not_equal)
        .set_age_field(compare: compare_symbol.contain, value: "16,17,18")
        .update_entity();
